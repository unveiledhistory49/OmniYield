### 3. Core Integration ¨C Gasless Deposit Button

Add this in your deposit component (`components/DepositForm.tsx`).

```tsx
import { useAccount, useWalletClient } from 'wagmi'
import { createPublicClient, http, parseAbi, getAddress } from 'viem'
import { base } from 'viem/chains'
import {
  createPimlicoPaymasterClient,
  createPimlicoBundlerClient,
  createSmartAccountClient,
  signerToSafeSmartAccount, // or toSimpleSmartAccount
  privateKeyToAccount, // only for testing
} from '@pimlicolabs/permissionless'
import { ENTRYPOINT_ADDRESS_V07 } from '@pimlicolabs/permissionless' // v0.7 entrypoint

// Your deployed ERC-4626 vault on Base
const VAULT_ADDRESS = '0xYourVaultAddressHere' as const

const apiKey = process.env.NEXT_PUBLIC_PIMLICO_API_KEY!
const pimlicoUrl = `https://api.pimlico.io/v2/base/rpc?apikey=${apiKey}`

// Public client (reuse your existing one)
const publicClient = createPublicClient({ chain: base, transport: http() })

// Pimlico clients
const paymasterClient = createPimlicoPaymasterClient({
  transport: http(pimlicoUrl),
  entryPoint: ENTRYPOINT_ADDRESS_V07,
})

const bundlerClient = createPimlicoBundlerClient({
  transport: http(pimlicoUrl),
  entryPoint: ENTRYPOINT_ADDRESS_V07,
})

// In your component (React hook)
export function GaslessDeposit({ amount }: { amount: bigint }) {
  const { address } = useAccount()
  const { data: walletClient } = useWalletClient()

  const handleGaslessDeposit = async () => {
    if (!walletClient || !address) return

    // 1. Create smart account from user's connected wallet (EOA as owner)
    const smartAccount = await signerToSafeSmartAccount(publicClient, {
      signer: walletClient, // user's wagmi walletClient
      entryPoint: ENTRYPOINT_ADDRESS_V07,
      safeVersion: '1.4.1',
    })

    // 2. Create smart account client with Pimlico middleware
    const smartAccountClient = createSmartAccountClient({
      account: smartAccount,
      chain: base,
      bundlerTransport: http(pimlicoUrl),
      middleware: {
        gasPrice: async () => (await bundlerClient.getUserOperationGasPrice()).fast,
        sponsorUserOperation: paymasterClient.sponsorUserOperation, // ¡û This makes it gasless!
      },
    })

    // 3. Build the deposit call to your ERC-4626 vault
    const depositCalldata = await publicClient.simulateContract({
      address: VAULT_ADDRESS,
      abi: parseAbi(['function deposit(uint256 assets, address receiver)']),
      functionName: 'deposit',
      args: [amount, address],
    })

    // 4. Send gasless!
    const txHash = await smartAccountClient.sendTransaction({
      to: VAULT_ADDRESS,
      data: depositCalldata.data,
      value: 0n,
    })

    console.log('Gasless deposit included!', `https://basescan.org/tx/${txHash}`)
  }

  return (
    <button onClick={handleGaslessDeposit} className="...">
      Deposit Gaslessly (Pimlico)
    </button>
  )
}
```

**That¡¯s it.** One button = gasless deposit. The user never sees a gas fee.

### 4. Two Paymaster Flavors (Choose or Offer Both)

**A. Verifying Paymaster (True Gasless ¨C Recommended for UX)**  
- You pay the gas from your Pimlico balance.  
- Users love it. Add in portfolio: ¡°Deposit with 0 ETH required¡±.  
- Control via Sponsorship Policies (only allow calls to your vault).

**B. ERC-20 Paymaster (User pays in USDC)**  
Add this to the `sendTransaction` call:

```ts

```

Users approve USDC once ¡ú gas paid in USDC. Perfect upsell.
