### 1. Live Vault TVL (Your Own Contracts)

**Base 每 ERC-4626 (add to `lib/onchain/base.ts`)**

```ts
import { createPublicClient, http, parseAbi } from 'viem'
import { base } from 'viem/chains'

const client = createPublicClient({
  chain: base,
  transport: http('https://rpc.ankr.com/base')   // or mainnet.base.org
})

export async function getBaseVaultTVL(vaultAddress: `0x${string}`) {
  const totalAssets = await client.readContract({
    address: vaultAddress,
    abi: parseAbi(['function totalAssets() view returns (uint256)']),
    functionName: 'totalAssets',
  })

  // Convert to USD (use DefiLlama price)
  const priceRes = await fetch(`https://api.llama.fi/coingecko/usdc`)
  const { price } = await priceRes.json()
  const tvlUsd = Number(totalAssets) * price / 1e6   // USDC decimals

  return { tvlUsd: Math.round(tvlUsd), totalAssets }
}
```

**Solana 每 Anchor Vault (add to `lib/onchain/solana.ts`)**

```ts
import { Connection, PublicKey } from '@solana/web3.js'

const connection = new Connection('https://mainnet.helius-rpc.com/?api-key=YOUR_FREE_HELIUS_KEY') 
// Get free key in 30 seconds at helius.dev

export async function getSolanaVaultTVL(vaultPda: string, underlyingMint: string) {
  const vaultInfo = await connection.getAccountInfo(new PublicKey(vaultPda))
  // Parse your Anchor account data (use your IDL)
  const underlyingBalance = await connection.getTokenAccountBalance(new PublicKey(underlyingMint)) // or your PDA

  const priceRes = await fetch('https://api.llama.fi/coingecko/solana')
  const { price } = await priceRes.json()

  const tvlUsd = Number(underlyingBalance.value.uiAmount) * price
  return { tvlUsd: Math.round(tvlUsd) }
}
```

### 3. Frontend 每 Live & Cached (TanStack Query Example)

```tsx
// hooks/useOmniYieldAnalytics.ts
import { useQuery } from '@tanstack/react-query'

export function useOmniYieldAnalytics() {
  return useQuery({
    queryKey: ['omniyield-analytics'],
    queryFn: async () => {
      const [baseTVL, solTVL, strategies] = await Promise.all([
        getBaseVaultTVL(BASE_VAULT),
        getSolanaVaultTVL(SOL_VAULT_PDA, SOL_UNDERLYING),
        getStrategyData()
      ])
      return {
        totalTVL: baseTVL.tvlUsd + solTVL.tvlUsd,
        baseTVL: baseTVL.tvlUsd,
        solTVL: solTVL.tvlUsd,
        strategies,
        lastUpdated: Date.now()
      }
    },
    refetchInterval: 15000,     // 15s refresh 每 looks insanely live
    staleTime: 10000,
  })
}
```

Use it anywhere:

```tsx
const { data } = useOmniYieldAnalytics()
<div>TVL: ${data?.totalTVL.toLocaleString()}</div>
```
