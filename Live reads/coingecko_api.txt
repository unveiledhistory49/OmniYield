### 2. Create the Price Utility (`lib/coingecko.ts`)

Create this file ˇŞ it handles everything (simple coins + contract-address tokens on Base & Solana).

```ts
// lib/coingecko.ts
const BASE_URL = 'https://pro-api.coingecko.com/api/v3';
const API_KEY = process.env.COINGECKO_API_KEY!;

const headers = {
  'x-cg-pro-api-key': API_KEY,
  'Content-Type': 'application/json',
};

// Batch major coins (SOL, JitoSOL, etc. by CoinGecko ID)
export async function getSimplePrices(coinIds: string[]) {
  const ids = coinIds.join(',');
  const res = await fetch(
    `${BASE_URL}/simple/price?ids=${ids}&vs_currencies=usd&include_24hr_change=true`,
    { headers, next: { revalidate: 30 } } // Auto-caches for 30s ˇŞ perfect UX
  );
  if (!res.ok) throw new Error('CoinGecko simple price failed');
  return res.json();
}

// Perfect for any token by contract/mint address (USDC on Base, mSOL on Solana, etc.)
export async function getTokenPriceByAddress(
  network: 'base' | 'solana' | 'ethereum',
  addresses: string | string[]
) {
  const addrStr = Array.isArray(addresses) ? addresses.join(',') : addresses;
  const res = await fetch(
    `${BASE_URL}/simple/token_price/${network}?contract_addresses=${addrStr}&vs_currencies=usd&include_24hr_change=true`,
    { headers, next: { revalidate: 30 } }
  );
  if (!res.ok) throw new Error('CoinGecko token price failed');
  return res.json();
}

// Bonus: On-chain DEX-sourced price (more accurate for low-liquidity tokens)
export async function getOnChainTokenPrice(network: 'base' | 'solana', address: string) {
  const res = await fetch(
    `${BASE_URL}/onchain/simple/networks/${network}/token_price/${address}`,
    { headers, next: { revalidate: 20 } }
  );
  return res.json();
}
```

import { getTokenPriceByAddress } from '@/lib/coingecko';
import { TOKENS } from '@/lib/constants/tokens';

export async function getBaseVaultTVL(vaultAddress: `0x${string}`) {
  const totalAssets = await publicClient.readContract({ /* your ERC-4626 totalAssets call */ });

  const priceData = await getTokenPriceByAddress('base', TOKENS.USDC_BASE);
  const usdcPrice = priceData[TOKENS.USDC_BASE.toLowerCase()]?.usd ?? 1;

  const tvlUsd = Number(totalAssets) * usdcPrice / 1_000_000; // USDC 6 decimals
  return { tvlUsd: Math.round(tvlUsd), totalAssets };
}
