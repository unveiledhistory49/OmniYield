Step-by-Step: One-Click “Deposit to Solana Vault from Base” (or vice versa)
1. Add the deBridge Script (once, in your root layout or deposit page)

// app/layout.tsx or components/DepositModal.tsx
'use client';

useEffect(() => {
  const script = document.createElement('script');
  script.src = 'https://app.debridge.com/assets/scripts/widget.js';
  script.async = true;
  document.body.appendChild(script);

  return () => { document.body.removeChild(script); };
}, []);

2. The Magic Component (copy-paste ready)

'use client';

import { useEffect, useState } from 'react';
import { useWallet } from '@solana/wallet-adapter-react'; // your existing Solana adapter
import { useAccount } from 'wagmi'; // your EVM wagmi

const DEBRIDGE_CONFIG = {
  element: 'debridge-crosschain',
  v: '1',
  mode: 'deswap',           // swap + bridge
  inputChain: 8453,         // Base
  outputChain: 50104,       // Solana (deBridge ID)
  inputCurrency: '0x833589fCD6eDb6E08f4c7C32D4f71b54bdA02913', // USDC on Base
  outputCurrency: 'EPjFWdd5AufqSSqeM2qN1xzybapC8G4wEGGkZwyTDt1v', // USDC on Solana
  theme: 'dark',
  width: '100%',
  height: '620',
  styles: btoa(JSON.stringify({
    appBackground: '#0a0a0a',
    appAccentBg: '#111111',
    primary: '#00ff9f',
    fontColor: '#ffffff',
    borderColor: '#222',
    borderRadius: 16,
  })),
};

export default function CrossChainDepositButton() {
  const { publicKey: solanaPubkey } = useWallet();
  const { address: evmAddress } = useAccount();
  const [showWidget, setShowWidget] = useState(false);

  const openWidget = () => {
    const userSolanaAddress = solanaPubkey?.toBase58();
    
    const config = {
      ...DEBRIDGE_CONFIG,
      // Auto-detect & prefill based on what user wants
      inputChain: 8453, // or dynamic from state
      outputChain: 50104,
      address: userSolanaAddress, // receiver = user's Solana wallet
      amount: '50', // optional prefill
    };

    if (window.deBridge) {
      window.deBridge.widget(config);
      setShowWidget(true);
    }
  };

  // Listen for successful bridge
  useEffect(() => {
    const handleBridgeSuccess = (event: any) => {
      if (event.detail?.status === 'completed') {
        console.log('✅ Funds arrived on Solana!', event.detail);
        
        // 🔥 Auto-trigger deposit into your OmniYield Solana vault
        if (solanaPubkey) {
          // Call your existing Anchor deposit function here
          // depositToVault(event.detail.amount, solanaPubkey);
          alert('Funds bridged! Depositing into OmniYield vault now...');
          // Show beautiful success modal with TVL update animation
        }
      }
    };

    window.addEventListener('debridge-event', handleBridgeSuccess);
    return () => window.removeEventListener('debridge-event', handleBridgeSuccess);
  }, [solanaPubkey]);

  return (
    <button 
      onClick={openWidget}
      className="w-full bg-gradient-to-r from-cyan-400 to-emerald-400 text-black font-bold py-4 rounded-2xl hover:scale-105 transition-all flex items-center justify-center gap-2"
    >
      🚀 Deposit to Solana Vault (from Base)
    </button>
  );
}
